import axios from 'axios';


const sendRequest = () => {
    
    // Формируем объект userData с данными пользователя
    const userData = {
        num_rooms: 2,  // количество комнат
        metro: 'Мякинино', // станция метро
        priceMin: 50000,  // минимальная цена
        priceMax: 70000  // максимальная цена
    };

    // Отправляем данные на бекенд в формате JSON
    axios.post('http://server:8000/process_data', userData)
      .then(response => {
        // Обработка успешного ответа от сервера, если это необходимо
        console.log('Данные успешно получены с бекенда!!:', response.data);
      })
      .catch(error => {
        // Обработка ошибок, если это необходимо
        console.error('Ошибка при получении данных:', error);
      });
};

sendRequest();
